         main(\x) = (0.5 * \x)^3;
         invmain(\x) = \x^(1/3) * 2;}]

  \draw[->] ({invmain(-6)}, -1.2)
         -- ({invmain(6)}, -1.2);

   {\draw (\x cm,-11.25 mm) -- (\x cm,-12.75 mm)
      node[below, text height=1.6ex]{$\xtext$};}

          variable=\x, samples=12, Cyan!35!black,

          smooth] plot({\x}, {sdrob(main(\x))});
  }
